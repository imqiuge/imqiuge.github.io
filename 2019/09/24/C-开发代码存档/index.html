    //核对数据库中的用户名与密码
    bool checkUserPassword(string sUser, string sPass)
    {
        String connsql = "server=120.79.140.200;database=HuaXia;user id=sa;pwd=Www.023**.cn;"; // 数据库连接字符串,database设置为自己的数据库名，以Windows身份验证
        //创建连接对象
        SqlConnection connection = new SqlConnection(connsql);

        string sql = string.Format("SELECT * FROM  fa_user WHERE username='{0}'", sUser);  // 查询员工类别的sql语句，也可以是增改删语句。

        // 设置command命令执行的语句
        SqlCommand command = new SqlCommand(sql, connection);
        try
        {
            connection.Open(); // 打开数据库连接
            SqlDataReader dataReader = command.ExecuteReader(); // 执行查询
            string sSqlpass = ""; //数据库中加密的密码
            string sSqlSalt = ""; //salt，用于判断密码
            while (dataReader.Read())
            {
                sSqlpass = (string)dataReader["password"];
                sSqlSalt = (string)dataReader["salt"];
            }
            if(sSqlpass.Length != 0)
            {
                string sText = EncryptWithMD5(EncryptWithMD5(sPass) + sSqlSalt);//MD5多次加密完
                if( sSqlpass == sText)  //对比方式
                {
                    //MessageBox.Show("对比成功");
                    return true;
                }
            }
            dataReader.Close();
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.Message);
            Console.WriteLine(ex.Message);
        }
        finally
        {
            connection.Close();
        }
        return false;
    }

//MD5加密
public static string EncryptWithMD5(string source)
{
byte[] sor = Encoding.UTF8.GetBytes(source);
MD5 md5 = MD5.Create();
byte[] result = md5.ComputeHash(sor);
StringBuilder strbul = new StringBuilder(40);
for (int i = 0; i < result.Length; i++)
{
strbul.Append(result[i].ToString(“x2”));//加密结果”x2”结果为32位,”x3”结果为48位,”x4”结果为64位

        }
        return strbul.ToString();
    }

    void killProcess()
    {
        Process[] pro = Process.GetProcesses();//获取已开启的所有进程

        //遍历所有查找到的进程

        for (int i = 0; i < pro.Length; i++)
        {

            //判断此进程是否是要查找的进程
            if (pro[i].ProcessName.ToString().ToLower() == "notepad")
            {
                pro[i].Kill();//结束进程
            }
        }
    }

    /// <summary>
    /// 处理部门的字符，因为有多级，以最后一个\为标志
    /// 原始输出的字符例：深圳市华夏技工学校\18届19届学生\18电商2班
    /// </summary>
    string CutGroup(string sGroup)
    {
        int nPos = sGroup.LastIndexOf(@"\");
        return sGroup.Substring(nPos + 1); ;
    }
